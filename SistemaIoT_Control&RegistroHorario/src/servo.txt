#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/ledc.h"
#include "esp_err.h"

// Configuración del Servo
#define SERVO_PIN           13          // GPIO donde conectas la señal
#define SERVO_LEDC_CHANNEL  LEDC_CHANNEL_0
#define SERVO_LEDC_TIMER    LEDC_TIMER_0
#define SERVO_LEDC_MODE     LEDC_LOW_SPEED_MODE

// Cálculos de Duty Cycle (Resolución de 13 bits: 0 a 8191)
// 50Hz -> periodo de 20ms. 
// Duty = (Tiempo_pulso / Periodo) * (2^Resolucion - 1)
#define SERVO_MIN_DUTY      205 // ~0.5ms (0°)
#define SERVO_MAX_DUTY      1024 // ~2.5ms (180°)

void init_servo() {
    // 1. Configurar el Timer del PWM
    ledc_timer_config_t ledc_timer = {
        .speed_mode       = SERVO_LEDC_MODE,
        .timer_num        = SERVO_LEDC_TIMER,
        .duty_resolution  = LEDC_TIMER_13_BIT,
        .freq_hz          = 50,  // Frecuencia estándar de servos
        .clk_cfg          = LEDC_AUTO_CLK
    };
    ledc_timer_config(&ledc_timer);

    // 2. Configurar el Canal
    ledc_channel_config_t ledc_channel = {
        .speed_mode     = SERVO_LEDC_MODE,
        .channel        = SERVO_LEDC_CHANNEL,
        .timer_sel      = SERVO_LEDC_TIMER,
        .intr_type      = LEDC_INTR_DISABLE,
        .gpio_num       = SERVO_PIN,
        .duty           = 0, 
        .hpoint         = 0
    };
    ledc_channel_config(&ledc_channel);
}

void set_servo_angle(uint32_t angle) {
    if (angle > 180) angle = 180;
    
    // Mapeo lineal de ángulo a Duty Cycle
    uint32_t duty = SERVO_MIN_DUTY + (((SERVO_MAX_DUTY - SERVO_MIN_DUTY) * angle) / 180);
    
    ledc_set_duty(SERVO_LEDC_MODE, SERVO_LEDC_CHANNEL, duty);
    ledc_update_duty(SERVO_LEDC_MODE, SERVO_LEDC_CHANNEL);
}

void app_main(void) {
    init_servo();
    printf("Servo MG996R inicializado.\n");

    while (1) {
        printf("Moviendo a 0 grados...\n");
        set_servo_angle(0);
        vTaskDelay(pdMS_TO_TICKS(2000));

        printf("Moviendo a 90 grados...\n");
        set_servo_angle(90);
        vTaskDelay(pdMS_TO_TICKS(2000));

        printf("Moviendo a 180 grados...\n");
        set_servo_angle(180);
        vTaskDelay(pdMS_TO_TICKS(2000));
    }
}
